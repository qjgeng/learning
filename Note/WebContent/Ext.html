                       Ext.Ajax.request({
                               url: this.application.contextPath + '/rest/charts/getMemberCommunityPatient',
                               scope: this,
                               params: {
                                      communityId: communityId,
                                      sourceId:currentCard.patientChart.tpId,
                                      patientId:currentCard.patientChart.patientId
                               },
                               success: function(response, opts)
                               {
                                      currentCard.el.unmask();
                                      var obj = Ext.JSON.decode(response.responseText), demographicForm,age,record;
                                      demographicForm = currentCard.down('#demographicForm');
                                      age=currentCard.down('#editDemographicAgelbl');
                                      //demographicForm.el.unmask();
                                      
                                      if(obj.success === true){
                                              record=Ext.create('cov.hfus.charts.ChartCard.model.MemberCommunityPatient',obj.patient || {});
                                              var languageField=record.data['language'];
                                             if(languageField && languageField!='en' && languageField!='es')
                                             {
                                                     record.data['language'] = 'other';
                                             }
                                              demographicForm.loadRecord(record);
                                              age.setText(record.data.age);
                                      currentCard.demographicsLoaded = true;
                                      this.updateCodeValueFields();
                                      this.processDemographicAfterLoad(currentCard,editable,true);
                                      }else{
                                              Ext.Msg.alert('Error', 'Error to load member community patient information.' + obj.errMsg);
                                      }
                           },
                           failure: function(response, opts) {
                               currentCard.el.unmask();
                               Ext.Msg.alert('Error', 'Failed to load member community  patient information. Please try again later.');
                           }
                       });

Ext.onReady(function(){
        Ext.define('Geng.model.MyData',{
               extend: 'Ext.data.Model',
               fields: [
                       'UserName',
                       'Sex',
                       'Age',
                       'No',
                       'Clz'
               ]              
        });
        
        var store = Ext.create('Ext.data.Store',{
               model: 'Geng.model.MyData',
               pageSize: 5,
               proxy:{
                       type: 'ajax',
                       url: 'mydata.json',
                       reader: {
                              type: 'json',
                              root: 'items'
                       }
               },
               autoLoad:true
        });
        
        
        var grid = Ext.create('Ext.grid.Panel',{
               store: store,
               columnLines: true,
               selModel: Ext.create('Ext.selection.CheckboxModel'),
               columns:[{
                       text:'基本信息',
                       columns: [
                       {
                              text:'姓名', width:120, dataIndex:'UserName',sortable:true
                       },
                       {
                              text:'性别', flex:1, dataIndex:'Sex',sortable:false
                       }
                       ]},
                       {
                              text:'年龄', width:100, dataIndex:'Age',sortable:true
                       },
                       {
                              text:'学号', width:100, dataIndex:'No',sortable:true
                       },
                       {
                              text:'班级', width:100, dataIndex:'Clz',sortable:true
                       }              
               ],

               renderTo: Ext.getBody(),
               viewConfig: {
                       stripeRows:true
               },
               style: {
                       'margin' : '20px'
               },
               bbar: Ext.create('Ext.PagingToolbar',{
                       store:store,
                       displayInfo:true,
                       displayMsg: '显示{0}-{1}条， 共计{2}条',
                       emptyMsg: '没有数据'
                       
               }),
               
               dockedItems:[

                              dock:'top',

                                             xtype:'button',

                                                             Ext.MessageBox.show({
                                                                    title:'提示',

                                                                    icon: Ext.MessageBox.WARNING
                                                             });

                                                             for(var i=0;i<records.length;i++) {
                                                                     cnt.push(records[i].get('UserName')+'-'+records[i].get('Clz'))
                                                             }
                                                             Ext.MessageBox.show({
                                                                    title:'提示',
                                                                    msg:cnt.join('<br/>'),
                                                                    icon: Ext.MessageBox.INFO
                                                             });
                                                     }
                                             },
                                             scope:this
                                      },'-',
                                      {
                                             width:300,
                                             fieldLabel:'搜索',
                                             labelWidth:50,
                                             xtype:'searchfield',
                                             store:store
                                      }
                              ]
                       }
               ]
               
        });
