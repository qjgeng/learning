$.fn.movieSlice = function(options) {
               var settings = $.extend({
                       movieElement: '#movie',
                       next: '#nextMovie',
                       prev: '#prevMovie',
                       first: '#firstMovie',
                       last: '#lastMovie',
                       thumbnailFolder: 'thumbnail',
                       movieFolder: 'movie'
               },options||{});
              
               settings.curIndex = 0;
              
               var thumbnails = this;
              
               $(this).each(function(pos){
                       $(this).data('index',pos);                  
               });
              
               $(this).on('click',function(event){
                       var index = $(this).data('index');
                       showPhoto(index);
               });
              
               function showPhoto(index) {
                       $(settings.movieElement).css({opacity:0.0});
                       $(settings.movieElement).attr('src',$(thumbnails)[index].src.replace(settings.thumbnailFolder,settings.movieFolder));
                       $(settings.movieElement).css({opacity:0.0}).animate({                     
                  opacity:1.0                           
            },500);   
            settings.curIndex = index;
               }
              
               $(settings.first).on('click',function(event){
                       settings.curIndex = 0;
                       showPhoto(settings.curIndex);               
               });
              
               $(settings.last).on('click',function(event){
                       settings.curIndex = thumbnails.length -1;
                       showPhoto(settings.curIndex);               
               });
              
               $(settings.next).on('click',function(event){
                       if(settings.curIndex != thumbnails.length -1) {
                              settings.curIndex ++;
                       }
                       showPhoto(settings.curIndex);               
               });
              
               $(settings.prev).on('click',function(event){
                       if(settings.curIndex != 0) {
                              settings.curIndex --;
                       }
                       showPhoto(settings.curIndex);               
               });
        };
